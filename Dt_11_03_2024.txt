create table student2(
htno number(5),
sname varchar2(10),
m1 number(3),
m2 number(3),
total number(4),
avgm number(3),
result varchar2(5),
class varchar2(10));




declare
phtno student2.htno%type;
psname student2.sname%type;
pm1 student2.m1%type;
pm2 student2.m2%type;
ptotal student2.total%type;
pavgm student2.avgm%type;
presult student2.result%type;
pclass student2.class%type;
begin
phtno:=&phtno;
psname:='&psname';
pm1:=&pm1;
pm2:=&pm2;

ptotal:=pm1+pm2;
pavgm:=ptotal/2;

if(pm1>35 ANd pm2>35) then
presult:='pass';
case
when pavgm>=70 then pclass:='Dist';
when pavgm>=60 AND pavgm<70 then pclass:='First';
when pavgm>=50 AND pavgm<60 then pclass:='Second';
else pclass:='Third';
end case;
else
presult:='fail';
pclass:=null;
end if;

insert into student2 values(phtno,psname,pm1,pm2,ptotal,pavgm,presult,pclass);

dbms_output.put_line('Inserted successfully...');
end;
